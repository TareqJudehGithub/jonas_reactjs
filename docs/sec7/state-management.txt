function handleItems(item) {
		/*
		 We cannot directly update the initialItems array, instead we create a new one.
		 In React, we cannot mutate the state, we cannot just push() the new item into the
		 initialItems array.
		*/
		setItems((items) => [...items, item]); // items = original initialItems array
	}

  Lifting up the state 
   - Lifting up the state, is simply moving a state from a child component to the parent
     one, so more than one child component can access it.

import "./index.css";

// Parent component
function App() {
	const [items, setItems] = useState([]);

	// Handlers
	// Update the state handler function, which we must also pass (as a prop) to
	// Form component, where we create new items.
	function handleAddItems(item) {
		setItems((items) => [...items, item]);
	}

	return (
		<div className="app">
			<Logo />
			{/* Naming Convection for naming handle props*/}
			<Form onAddItems={handleAddItems} />
			<PackingList items={items} />
			<Stats />
		</div>
	);
}
// Child components
function Logo() {
	return <h1>🌴Far Away🎒</h1>;
}

// The form component purpose is to add new items into the array (update the state), but
// not to render it (the UI).
function Form({ onAddItems }) {
	// States
	const [description, setDescription] = useState("");
	const [quantity, setQuantity] = useState(1);

	// Handlers
	function handleSubmit(e) {
		e.preventDefault();

		// form inputs validation
		if (!description) {
			alert("Enter item name!");
		}

		const newItem = { id: Date.now(), description, quantity, packed: false };
		onAddItems(newItem);

		// Setting the description state back to its initial state.
		setDescription("");
		setQuantity(1);
	}

	return (
		<form className="add-form" onSubmit={handleSubmit}>
			<h3>What do you need for your 😍 trip?</h3>
			<select
				value={quantity}
				onChange={(e) => setQuantity(Number(e.target.value))}
			>
				{Array.from({ length: 20 }, (value, index) => index + 1).map((num) => (
					<option key={num} value={num}>
						{num}
					</option>
				))}
			</select>

			<input
				type="text"
				value={description}
				onChange={(e) => setDescription(e.target.value)}
				placeholder="item..."
				autoFocus
			/>
			<button>Add</button>
		</form>
	);
}

function PackingList({ items }) {
	return (
		<div className="list">
			<ul>
				{items.map((item) => (
					<Item key={item.id} listItems={item} />
				))}
			</ul>
		</div>
	);
}

function Item({ listItems }) {
	const { description, quantity, packed } = listItems;
	return (
		<li>
			<span
				style={packed ? { textDecoration: "line-through" } : { color: "red" }}
			>
				{quantity} {description}
			</span>
			<button>❌</button>
		</li>
	);
}

function Stats() {
	return (
		<footer className="stats">
			You have (X) item(s) on your list, and you already packed (X%)
		</footer>
	);
}

export default App;
